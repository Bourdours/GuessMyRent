CREATE TABLE USER_(
   id_user INT AUTO_INCREMENT,
   email VARCHAR(50)  NOT NULL,
   password VARCHAR(112)  NOT NULL,
   pseudo VARCHAR(50) ,
   avatar VARCHAR(50) ,
   is_admin BOOLEAN,
   PRIMARY KEY(id_user),
   UNIQUE(email)
);

CREATE TABLE MESSAGE(
   Id_message INT AUTO_INCREMENT,
   email VARCHAR(50)  NOT NULL,
   content VARCHAR(500)  NOT NULL,
   id_user INT NOT NULL,
   PRIMARY KEY(Id_message),
   FOREIGN KEY(id_user) REFERENCES USER_(id_user)
);

CREATE TABLE TYPE(
   id_type INT AUTO_INCREMENT,
   label VARCHAR(50)  NOT NULL,
   PRIMARY KEY(id_type)
);

CREATE TABLE STATUS(
   id_status INT AUTO_INCREMENT,
   label VARCHAR(50)  NOT NULL,
   PRIMARY KEY(id_status)
);

CREATE TABLE ESTATE(
   id_estate INT AUTO_INCREMENT,
   rent INT NOT NULL,
   is_charges_included BOOLEAN,
   adress VARCHAR(50) ,
   city VARCHAR(50)  NOT NULL,
   postcode INT NOT NULL,
   gps_y DECIMAL(15,8)  ,
   gps_x DECIMAL(15,8)  ,
   square_meters DECIMAL(15,2)   NOT NULL,
   room INT,
   chamber INT,
   floor_ INT,
   description VARCHAR(50) ,
   image1 VARCHAR(50)  NOT NULL,
   image2 VARCHAR(50) ,
   image3 VARCHAR(50) ,
   image4 VARCHAR(50) ,
   id_status INT NOT NULL,
   id_type INT NOT NULL,
   id_user INT NOT NULL,
   PRIMARY KEY(id_estate),
   FOREIGN KEY(id_status) REFERENCES STATUS(id_status),
   FOREIGN KEY(id_type) REFERENCES TYPE(id_type),
   FOREIGN KEY(id_user) REFERENCES USER_(id_user)
);

CREATE TABLE GAME(
   id_game INT AUTO_INCREMENT,
   game_result DECIMAL(15,2)  ,
   date_ DATETIME,
   guess INT NOT NULL,
   id_user INT,
   id_estate INT NOT NULL,
   PRIMARY KEY(id_game),
   FOREIGN KEY(id_user) REFERENCES USER_(id_user),
   FOREIGN KEY(id_estate) REFERENCES ESTATE(id_estate)
);

CREATE TABLE ESTATE_MODIFICATION(
   id_modification INT AUTO_INCREMENT,
   date_ DATETIME,
   modification VARCHAR(250) ,
   id_estate INT NOT NULL,
   id_user INT NOT NULL,
   PRIMARY KEY(id_modification),
   FOREIGN KEY(id_estate) REFERENCES ESTATE(id_estate),
   FOREIGN KEY(id_user) REFERENCES USER_(id_user)
);
